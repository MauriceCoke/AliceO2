48 template <typename Track>
49 bool GetImpactParameterAndError(const Track& track, const o2df::VertexBase& primaryVertex, double& impactParameter,
50  double& impactParameterError)
51 {
52  impactParameter = -999.;
53  impactParameterError = -999.;
54 
55  o2df::DCA dca;
56  // FIXME: get this from CCDB
57  constexpr float magneticField{5.0}; // in kG
58  auto trackParameter = getTrackParCov(track);
59  bool propagate = trackParameter.propagateToDCA(primaryVertex, magneticField, &dca);
60 
61  if (propagate) {
62  impactParameter = 1000 * dca.getY();
63  impactParameterError = 1000 * std::sqrt(dca.getSigmaY2());
64  }
65  return propagate;
66 }

178  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
179  const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
180  const o2::aod::McParticles& mcParticles, const o2::aod::McCollisions& mcCollisions)
181  {
182  const o2df::VertexBase primaryVertex = getPrimaryVertex(collision);
183 
184  for (const auto& track : tracks) {
185  const double deltaPt = track.label().pt() - track.pt();
186  ptDiffMCRec->Fill(deltaPt);
187 
188  const double deltaPtOverPt = deltaPt / track.pt();
189  ptResolution->Fill((deltaPtOverPt));
190  ptResolutionVsPt->Fill(track.pt(), abs(deltaPtOverPt));
191  ptResolutionVsEta->Fill(track.eta(), abs(deltaPtOverPt));
192 
193  const double deltaEta = track.label().eta() - track.eta();
194  etaDiffMCRec->Fill(deltaEta);
195  etaDiffMCRecoVsEtaMC->Fill(deltaEta, track.label().eta());
196  etaDiffMCRecoVsEtaReco->Fill(deltaEta, track.eta());
197 
198  const auto deltaPhi = track_utils::ConvertPhiRange(track.label().phi() - track.phi());
199  phiDiffMCRec->Fill(deltaPhi);
200 
201  double impactParameter{-999.};
202  double impactParameterError{-999.};
203 
204  const bool propagate =
205  track_utils::GetImpactParameterAndError(track, primaryVertex, impactParameter, impactParameterError);
206 
207  if (propagate) {
208  impactParameterVsPt->Fill(track.pt(), impactParameter);
209  impactParameterVsEta->Fill(track.eta(), impactParameter);
210  impactParameterErrorVsPt->Fill(track.pt(), impactParameterError);
211  impactParameterErrorVsEta->Fill(track.eta(), impactParameterError);
212  }
213  }
214  }
215 };
