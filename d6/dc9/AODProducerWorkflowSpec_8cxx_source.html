44 void AODProducerWorkflowDPL::findMinMaxBc(gsl::span<const o2::ft0::RecPoints>& ft0RecPoints, gsl::span<const o2::dataformats::TrackTPCITS>& tracksITSTPC)
45 {
46  for (auto& ft0RecPoint : ft0RecPoints) {
47  uint64_t bc = ft0RecPoint.getInteractionRecord().orbit * o2::constants::lhc::LHCMaxBunches + ft0RecPoint.getInteractionRecord().bc;
48  if (minGlBC > bc) {
49  minGlBC = bc;
50  }
51  if (maxGlBC < bc) {
52  maxGlBC = bc;
53  }
54  }
55 
56  for (auto& trackITSTPC : tracksITSTPC) {
57  Double_t timeStamp = trackITSTPC.getTimeMUS().getTimeStamp() * 1.E3; // ms to ns
58  uint64_t bc = (int)(timeStamp / o2::constants::lhc::LHCBunchSpacingNS);
59  if (minGlBC > bc) {
60  minGlBC = bc;
61  }
62  if (maxGlBC < bc) {
63  maxGlBC = bc;
64  }
65  }
66 }

69 void AODProducerWorkflowDPL::fillTracksTable(const TracksType& tracks, std::vector<int>& vCollRefs, const TracksCursorType& tracksCursor, int trackType)
70 {
71  for (int i = 0; i < tracks.size(); i++) {
72  auto& track = tracks[i];
73  int collisionID = vCollRefs[i];
74  tracksCursor(0,
75  collisionID,
76  trackType,
77  track.getX(),
78  track.getAlpha(),
79  track.getY(),
80  track.getZ(),
81  track.getSnp(),
82  track.getTgl(),
83  track.getQ2Pt(),
84  TMath::Sqrt(track.getSigmaY2()),
85  TMath::Sqrt(track.getSigmaZ2()),
86  TMath::Sqrt(track.getSigmaSnp2()),
87  TMath::Sqrt(track.getSigmaTgl2()),
88  TMath::Sqrt(track.getSigma1Pt2()),
89  (Char_t)(128. * track.getSigmaZY() / track.getSigmaZ2() / track.getSigmaY2()),
90  (Char_t)(128. * track.getSigmaSnpY() / track.getSigmaSnp2() / track.getSigmaY2()),
91  (Char_t)(128. * track.getSigmaSnpZ() / track.getSigmaSnp2() / track.getSigmaZ2()),
92  (Char_t)(128. * track.getSigmaTglY() / track.getSigmaTgl2() / track.getSigmaY2()),
93  (Char_t)(128. * track.getSigmaTglZ() / track.getSigmaTgl2() / track.getSigmaZ2()),
94  (Char_t)(128. * track.getSigmaTglSnp() / track.getSigmaTgl2() / track.getSigmaSnp2()),
95  (Char_t)(128. * track.getSigma1PtY() / track.getSigma1Pt2() / track.getSigmaY2()),
96  (Char_t)(128. * track.getSigma1PtZ() / track.getSigma1Pt2() / track.getSigmaZ2()),
97  (Char_t)(128. * track.getSigma1PtSnp() / track.getSigma1Pt2() / track.getSigmaSnp2()),
98  (Char_t)(128. * track.getSigma1PtTgl() / track.getSigma1Pt2() / track.getSigmaTgl2()));
99  }

115 void AODProducerWorkflowDPL::run(ProcessingContext& pc)
116 {
117  mTimer.Start(false);
118 
119  auto ft0ChData = pc.inputs().get<gsl::span<o2::ft0::ChannelDataFloat>>("ft0ChData");
120  auto ft0RecPoints = pc.inputs().get<gsl::span<o2::ft0::RecPoints>>("ft0RecPoints");
121  auto primVer2TRefs = pc.inputs().get<gsl::span<o2::vertexing::V2TRef>>("primVer2TRefs");
122  auto primVerGIs = pc.inputs().get<gsl::span<o2::vertexing::GIndex>>("primVerGIs");
123  auto primVertices = pc.inputs().get<gsl::span<o2::vertexing::PVertex>>("primVertices");
124  auto tracksITS = pc.inputs().get<gsl::span<o2::its::TrackITS>>("trackITS");
125  auto tracksITSTPC = pc.inputs().get<gsl::span<o2::dataformats::TrackTPCITS>>("tracksITSTPC");
126  auto tracksTPC = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("trackTPC");
127 
128  LOG(INFO) << "FOUND " << tracksTPC.size() << " TPC tracks";
129  LOG(INFO) << "FOUND " << tracksITS.size() << " ITS tracks";
130  LOG(INFO) << "FOUND " << tracksITSTPC.size() << " ITCTPC tracks";
131 
132  auto& bcBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "BC"});
133  auto& collisionsBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "COLLISION"});
134  auto& ft0Builder = pc.outputs().make<TableBuilder>(Output{"AOD", "FT0"});
135  auto& mcCollisionsBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "MCCOLLISION"});
136  auto& tracksBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "TRACK"});
137 
138  // TODO:
139  // add FV0A, FV0C, FDD tables
140  // auto& fv0aBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "FV0A"});
141  // auto& fv0cBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "FV0C"});
142  // auto& fddBuilder = pc.outputs().make<TableBuilder>(Output{"AOD", "FDD"});
143 
144  auto bcCursor = bcBuilder.cursor<o2::aod::BCs>();
145  auto collisionsCursor = collisionsBuilder.cursor<o2::aod::Collisions>();
146  auto ft0Cursor = ft0Builder.cursor<o2::aod::FT0s>();
147  auto mcCollisionsCursor = mcCollisionsBuilder.cursor<o2::aod::McCollisions>();
148  auto tracksCursor = tracksBuilder.cursor<o2::aodproducer::TracksTable>();
149 
150  // TODO:
151  // add FV0A, FV0C, FDD tables
152  // auto fv0aCursor = fv0aBuilder.cursor<o2::aod::FV0As>();
153  // auto fv0cCursor = fv0cBuilder.cursor<o2::aod::FV0Cs>();
154  // auto fddCursor = fddBuilder.cursor<o2::aod::FDDs>();
155 
156  // float dummyfv0AmplA[48] = {0.};
157  // float dummyfv0AmplC[32] = {0.};
158  // float dummyfddAmplA[4] = {0.};
159  // float dummyfddAmplC[4] = {0.};
160  // fv0aCursor(0, (uint64_t)0, dummyfv0AmplA, 0.f, (uint8_t)0);
161  // fv0cCursor(0, (uint64_t)0, dummyfv0AmplC, 0.f);
162  // fddCursor(0, (uint64_t)0, dummyfddAmplA, dummyfddAmplC, 0.f, 0.f, (uint8_t)0);
163 
164  o2::steer::MCKinematicsReader mcReader("collisioncontext.root");
165  const auto context = mcReader.getDigitizationContext();
166  const auto& records = context->getEventRecords();
167  const auto& parts = context->getEventParts();
168 
169  LOG(INFO) << "FOUND " << records.size() << " records";
170  LOG(INFO) << "FOUND " << parts.size() << " parts";
171 
172  // TODO:
173  // change index to BCId (?)
174  // figure out generatorID and collision weight
175  int index = 0;
176  int generatorID = 0;
177  float mcColWeight = 1.;
178 
179  // filling mcCollison table
180  for (auto& rec : records) {
181  auto time = rec.getTimeNS();
182  auto& colparts = parts[index];
183  auto eventID = colparts[0].entryID;
184  auto sourceID = colparts[0].sourceID;
185  auto& header = mcReader.getMCEventHeader(sourceID, eventID);
186  mcCollisionsCursor(0,
187  index,
188  generatorID,
189  header.GetX(),
190  header.GetY(),
191  header.GetZ(),
192  time,
193  mcColWeight,
194  header.GetB());
195  index++;
196  }
197 
198  findMinMaxBc(ft0RecPoints, tracksITSTPC);
199 
200  std::map<uint64_t, uint64_t> mGlobBC2BCID;
201 
202  // TODO:
203  // get real run number and triggerMask
204  int runNumber = 244918;
205  uint64_t triggerMask = 1;
206 
207  // filling BC table and map<globalBC, BCId>
208  for (uint64_t i = 0; i <= maxGlBC - minGlBC; i++) {
209  bcCursor(0,
210  runNumber,
211  minGlBC + i,
212  triggerMask);
213  mGlobBC2BCID[minGlBC + i] = i;
214  }
215 
216  // vector of FT0 amplitudes
217  std::vector<float> vAmplitudes(208, 0.);
218  // filling FT0 table
219  for (auto& ft0RecPoint : ft0RecPoints) {
220  const auto channelData = ft0RecPoint.getBunchChannelData(ft0ChData);
221  // TODO:
222  // switch to calibrated amplitude
223  for (auto& channel : channelData) {
224  vAmplitudes[channel.ChId] = channel.QTCAmpl; // amplitude, mV
225  }
226  float aAmplitudesA[96];
227  float aAmplitudesC[112];
228  std::copy(vAmplitudes.begin(), vAmplitudes.begin() + 95, aAmplitudesA);
229  std::copy(vAmplitudes.begin() + 96, vAmplitudes.end(), aAmplitudesC);
230  uint64_t globalBC = ft0RecPoint.getInteractionRecord().orbit * o2::constants::lhc::LHCMaxBunches + ft0RecPoint.getInteractionRecord().bc;
231  ft0Cursor(0,
232  mGlobBC2BCID[globalBC],
233  aAmplitudesA,
234  aAmplitudesC,
235  ft0RecPoint.getCollisionTimeA() / 1E3, // ps to ns
236  ft0RecPoint.getCollisionTimeC() / 1E3, // ps to ns
237  ft0RecPoint.getTrigger().triggersignals);
238  }
239 
240  // initializing vectors for trackID --> collisionID connection
241  std::vector<int> vCollRefsITS(tracksITS.size(), -1);
242  std::vector<int> vCollRefsTPC(tracksTPC.size(), -1);
243  std::vector<int> vCollRefsTPCITS(tracksITSTPC.size(), -1);
244 
245  // filling collisions table
246  int collisionID = 0;
247  for (auto& vertex : primVertices) {
248  auto& cov = vertex.getCov();
249  auto& timeStamp = vertex.getTimeStamp();
250  Double_t tsTimeStamp = timeStamp.getTimeStamp() * 1E3; // ms to ns
251  uint64_t globalBC = tsTimeStamp / o2::constants::lhc::LHCBunchSpacingNS;
252  int BCid = mGlobBC2BCID[globalBC];
253  // TODO:
254  // get real collision time mask
255  int collisionTimeMask = 1;
256  collisionsCursor(0,
257  BCid,
258  vertex.getX(),
259  vertex.getY(),
260  vertex.getZ(),
261  cov[0],
262  cov[1],
263  cov[2],
264  cov[3],
265  cov[4],
266  cov[5],
267  vertex.getChi2(),
268  vertex.getNContributors(),
269  timeStamp.getTimeStamp(),
270  timeStamp.getTimeStampError(),
271  collisionTimeMask);
272  auto trackRef = primVer2TRefs[collisionID];
273  int start = trackRef.getFirstEntryOfSource(0);
274  int ntracks = trackRef.getEntriesOfSource(0);
275  for (int ti = 0; ti < ntracks; ti++) {
276  auto trackIndex = primVerGIs[start + ti];
277  const auto source = trackIndex.getSource();
278  // setting collisionID for tracks attached to vertices
279  if (source == o2::vertexing::GIndex::Source::TPC) {
280  vCollRefsTPC[trackIndex.getIndex()] = collisionID;
281  } else if (source == o2::vertexing::GIndex::Source::ITS) {
282  vCollRefsITS[trackIndex.getIndex()] = collisionID;
283  } else if (source == o2::vertexing::GIndex::Source::TPCITS) {
284  vCollRefsTPCITS[trackIndex.getIndex()] = collisionID;
285  } else {
286  LOG(WARNING) << "Unsupported track type!";
287  }
288  }
289  collisionID++;
290  }
291 
292  // filling tracks tables
293  if (mFillTracksITS) {
294  fillTracksTable(tracksITS, vCollRefsITS, tracksCursor, o2::vertexing::GIndex::Source::ITS); // fTrackType = 1
295  }
296  if (mFillTracksTPC) {
297  fillTracksTable(tracksTPC, vCollRefsTPC, tracksCursor, o2::vertexing::GIndex::Source::TPC); // fTrackType = 2
298  }
299  if (mFillTracksITSTPC) {
300  fillTracksTable(tracksITSTPC, vCollRefsTPCITS, tracksCursor, o2::vertexing::GIndex::Source::TPCITS); // fTrackType = 0
301  }
302 
303  mTimer.Stop();
304 }

341  outputs,
342  AlgorithmSpec{adaptFromTask<AODProducerWorkflowDPL>()},

347 }
